#### Problem
No. 437: https://leetcode.com/problems/path-sum-iii/

#### Code
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def pathSum(self, root: TreeNode, sum: int) -> int:
        self.answer = 0
        self.dfs(root, sum)
        return self.answer

    def dfs(self, node, sum):
        if not node:
            return

        self.dfs(node.left, sum)
        self.break_down(node, sum)
        self.dfs(node.right, sum)
    
    def break_down(self, node, sum):
        if not node:
            return
        if node.val == sum:
            self.answer += 1
        
        self.break_down(node.left, sum - node.val)
        self.break_down(node.right, sum - node.val)
```

class MinStack:

    def __init__(self):
        """
        initialize your data structure here.
        """
        self.elements = []
        self.min_element = float('inf')
        self.top_element_index = -1

    def push(self, x: int) -> None:
        self.elements.append(x)
        self.top_element_index += 1
        self.min_element = min(x, self.min_element)

    def pop(self) -> None:
        self.top_element_index -= 1
        
        top_element = self.elements.pop()
        if top_element == self.min_element:
            self.min_element = min(self.elements) if self.elements else float('inf')
        
        return top_element

    def top(self) -> int:
        return self.elements[self.top_element_index]

    def getMin(self) -> int:
        return self.min_element
